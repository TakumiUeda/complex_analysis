\end{Definition}

\subsection{コーシー・リーマンの方程式}
\begin{Theorem}[コーシー・リーマンの方程式]
\ \\ \par
複素変数$z=x+iy$の関数$f(z)=u(x,y)+iv(x,y)$について
\begin{align}
\begin{split}
\dfrac{\partial u}{\partial x} &= \dfrac{\partial v}{\partial y} \\
\dfrac{\partial u}{\partial y} &= -\dfrac{\partial v}{\partial x}
\end{split}
\end{align}
をコーシー・リーマンの方程式という．
\end{Theorem}

\section{コーシーの積分公式}
\begin{Theorem}[コーシーの積分公式]
\ \\ \par
$単連結領域内$D$f(z)$で正則である$f(z)$について，$D$のジョルダン閉曲線上を正の方向に$1$周する積分路を$C$とすると，$C$内部の任意の点$z$に関して
\begin{align}
\begin{split}
f(z) = \frac{1}{2 \pi i} \oint_{c} \frac{f(\zeta)}{\zeta -z} d \zeta
\end{split}
\end{align}
が成り立つ
\end{Theorem}
\begin{Proof}
\ \\ \par
$f(z)$は領域内で正則であるので$\zeta - $平面上で$C$の内部にある$\dfrac{f(\zeta)}{\zeta-z}$の特異点は$\zeta = z$だけであり，その時の留数は
\begin{align}
\begin{split}
\mbox{Res}(z) &= \lim_{\zeta \to z} (\zeta - z) \frac{f(\zeta)}{\zeta -z} d \zeta \\
&=f(z)
\end{split}
\end{align}
\end{Proof}

\section{ローラン展開}
\begin{Theorem}[ローラン級数]
\ \\ \par
関数$f(z)$を領域$0  \leq R_{1} < |z-a| <R_{2} $で正則とする．このとき，$f(z)$は点$a$を中心を中心とするローラン級数
\begin{align}
\begin{split}
f(z) = \sum_{n=0}^{\infty} a_{n} (z-a)^{n} + \sum_{n=1}^{\infty} \dfrac{a_{-n}}{(z-a)^{n}}\\[1.5ex]
a_{n} = \frac{1}{2 \pi i} \int_{|\zeta - a|=r} \dfrac{f(\zeta)}{(\zeta - a)^{n+1}} d\zeta \\[1.5ex] (n=0\pm1,\pm2, \cdots,0  \leq R_{1} < |z-a| <R_{2})
\end{split}
\end{align}
に展開できる．
\end{Theorem}

\section{留数と留数定理}
\begin{Theorem}[1位の極]
\ \\ \par
$a$が$1$位の極である場合に留数を求めるには
\begin{align}
\begin{split}
\mbox{Res}(a,f) &= \lim_{z \to a} (z-a) f(z)
\end{split}
\end{align}
\end{Theorem}

\begin{Theorem}[2位の極]
\ \\ \par
$a$が$2$位の極である場合に留数を求めるには
\begin{align}
\begin{split}
\mbox{Res}(a,f) &= \lim_{z \to a} \dfrac{d}{dz} \{ (z-a^{2}) f(z) \}
\end{split}
\end{align}
\end{Theorem}

\begin{Theorem}[n位の極]
\ \\ \par
$a$が$n$位の極である場合に留数を求めるには
\begin{align}
\begin{split}
\mbox{Res} (a,f) &= \dfrac{1}{(n-1)!} \lim_{z \to a} \dfrac{d^{n-1}}{dz^{n-1}} \{ (z-a)^{n} f(z) \}
\end{split}
\end{align}
\end{Theorem}
